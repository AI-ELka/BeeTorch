Active Sets : SELECT model_id,poison,poison_rate,max(date) > now()-interval 10 minute FROM logs GROUP BY model_id,poison,poison_rate;

Global Overview : SELECT table1.model_id,dataset,name,dimension,table1.accuracy,min(table1.epochs) as optimal_epoch,max.max_epochs as total_epochs,max.max_epochs-min(table1.epochs) as dif_epochs,table1.poison,table1.poison_rate FROM logs as table1,(SELECT model_id,poison,poison_rate,max(accuracy) as accuracy,max(epochs) as max_epochs FROM logs GROUP BY model_id,poison,poison_rate) as max, models WHERE table1.accuracy=max.accuracy and table1.model_id=max.model_id and table1.poison=max.poison and table1.poison_rate=max.poison_rate and table1.model_id=models.id GROUP BY table1.model_id,table1.accuracy,table1.poison,table1.poison_rate ORDER BY poison,table1.model_id;
